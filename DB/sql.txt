CREATE FUNCTION `get_doctor_name`(`d_id` varchar(45)) RETURNS varchar(500) CHARSET latin1
BEGIN
declare d_nam varchar(500);
set d_nam:=(select ifnull(d_name,"") from d_doctor_info where doctor_code=d_id and record_status='A' );
RETURN d_nam;
END
